Taint the Old node Group 
root@56bcca239888:~/ops-tools/eks_eksctl# kubectl taint nodes -l "alpha.eksctl.io/nodegroup-name=eks-poc-ng" migration=eks-poc-ng:NoSchedule
node/ip-10-0-8-105.ec2.internal tainted

Remove a taint from  specific node groups 

kubectl taint nodes -l "alpha.eksctl.io/nodegroup-name=eks-poc-ng" migration=eks-poc-ng:NoSchedule-

If we want to remove taint from all nodes  then 
kubectl taint nodes -all "alpha.eksctl.io/nodegroup-name=eks-poc-ng" migration=eks-poc-ng:NoSchedule-

root@56bcca239888:~/ops-tools/eks_eksctl# kubectl taint nodes -l "alpha.eksctl.io/nodegroup-name=eks-poc-ng" migration=eks-poc-ng:NoSchedule-
node/ip-10-0-8-105.ec2.internal untainted
